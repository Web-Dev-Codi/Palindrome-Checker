const palindromChecker = (str) => {
	let palindromEl = '';
	for (let i = 0; i < str.length; i++) {
		for (let j = 0; j < str.length; j++) {
			let str1 = Array.from(str[i])
			let str2 = Array.from(str[j])
            if (str1.includes(i) === str2.includes(j)) {
              palindromEl = str1.toString(str2)
            }
		}
	}
    return palindromEl;
};
